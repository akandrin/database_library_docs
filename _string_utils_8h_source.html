54 if (sizeof...(args) != GetSubstringsCount(_format, "{}")) {
55 assert(false);

57 }

63 ReplaceAll(format, "{}", "%s");
64
65 int sizeInt =
66 std::snprintf(nullptr, 0, format.c_str(), ToString(args).c_str()...);
67 if (sizeInt <= 0) {
68 assert(false);
69 return {};
70 }

75 // Не берем '\0'

77}
